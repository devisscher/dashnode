exports.gitPullRepository = function(gitRepo, domainName) {
    
    var exec = require('child_process').exec;
    exec('mkdir /var/www/' + domainName, function(error, stdout, stderr) {
        if (error) {
            console.log(error)
        } else {
            console.log(stdout);
        }
    });
    var exec2 = require('child_process').exec;
    var script = "cd /var/www/" + domainName;
    var script2 = " git init"
    exec2(script +  script2, function(error, stdout, stderr) {
        if (error) {
            console.log(error)
        } else {
            console.log(stdout);
        }
    });
    
}
